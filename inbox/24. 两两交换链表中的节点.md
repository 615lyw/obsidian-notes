// 递归：给一个链表的头结点，两两交换后返回新头结点
public ListNode swapPairs(ListNode head) {
    if (head == null || head.next == null) {
        return head;
    }
    // 剩余节点数大于 2
    ListNode node = swapPairs(head.next.next);
    ListNode newHead = head.next;
    newHead.next = head;
    head.next = node;
    return newHead;
}
