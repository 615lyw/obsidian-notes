class MyLinkedList {
    private Node head;
    private Node tail;
    private int size;

    public MyLinkedList() {
        head = new Node(-99);
        tail = new Node(-99);
        head.next = tail;
        tail.pre = head;
        size = 0;
    }

    private void debug() {
        for (Node i = head.next; i != tail; i = i.next) {
            System.out.print(i.val);
        }
        System.out.println();
    }
    
    public int get(int index) {
        Node node = getNode(index);
        //debug();
        return node == null ? -1 : node.val;
    }
    
    public void addAtHead(int val) {
        Node node = new Node(val);
        node.next = head.next;
        node.next.pre = node;
        node.pre = head;
        head.next = node;
        size++;
        //debug();
    }
    
    public void addAtTail(int val) {
        Node node = new Node(val);
        tail.pre.next = node;
        node.pre = tail.pre;
        node.next = tail;
        tail.pre = node;
        size++;
        //debug();
    }
    
    public void addAtIndex(int index, int val) {
        if (index > size) return;
        if (index < 0) {
            addAtHead(val);
            return;
        }
        if (index == size) {
            addAtTail(val);
            return;
        }

        Node node = getNode(index);
        Node newNode = new Node(val);
        newNode.pre = node.pre;
        node.pre.next = newNode;
        newNode.next = node;
        node.pre = newNode;
        size++;
        //debug();
    }
    
    public void deleteAtIndex(int index) {
        Node node = getNode(index);
        if (node == null) return;
        node.pre.next = node.next;
        node.next.pre = node.pre;
        node.next = null;
        node.pre = null;
        size--;
        //debug();
    }

    // 优化
    private Node getNode(int index) {
        if (index < 0 || index >= size) {
            return null;
        }
        Node cur = head;
        if (index + 1 < size - index) {
            // head 出发
            for (int i = 0; i <= index; i++) {
                cur = cur.next;
            }
        } else {
            // tail 出发
            cur = tail;
            for (int i = 0; i < size - index; i++) {
                cur = cur.pre;
            }
        }
        //System.out.println("index=" + index + ", cur.val=" + cur.val);
        return cur;
    }

    private static class Node {
        Node pre;
        Node next;
        int val;

        Node(int val) {
            this.val = val;
        }
    }
}
