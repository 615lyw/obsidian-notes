class Solution {
    public int removeElement(int[] nums, int val) {
        int left = 0; // 指向值不为 val 的元素应该放的位置
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != val) {
                nums[left] = nums[i];
                left++;
            }
        }
        return left;
    }
}
